% ---------------------------------------------------------------- 6. Why hard
\begin{frame}{Why is this hard? Physics doesn't do retries}
	\begin{itemize}
		\setlength\itemsep{0.75em}
		\item \textbf{Physics won't wait.} A web request can take a second longer; smoke spreading
		through a corridor cannot. \hilite{Deadlines are physical, not polite.}
		\item \textbf{Sensors lie.} Drift, noise, a radiator under the thermometer.
		Your software's entire world-view arrives through \hilite{imperfect measurements}.
		\item \textbf{Actuators are slow and irreversible.} Heating a room takes an hour;
		\subtle{unsending} a sprinkler activation takes a mop.
		\item \textbf{Failures are physical.} A crashed e-commerce pod loses a shopping cart.
		A crashed ventilation controller fills a meeting room with CO$_2$.
		\item \textbf{You can't pause the world to debug it.} The building keeps evolving while
		your process is down — \hilite{state changes whether you observe it or not}.
	\end{itemize}
\end{frame}

% ---------------------------------------------------------------- 8. From cloud to edge
\begin{frame}{From cloud to edge: four forces, one direction}
	\begin{itemize}
		\setlength\itemsep{0.7em}
		\item \textbf{Latency.} Cloud round-trip: 50--300\,ms, sometimes worse. Fine for email —
		not for a fire damper. \hilite{Physical processes cannot wait for a distant datacenter.}
		\item \textbf{Bandwidth.} Thousands of sensors $\times$ readings every second — shipping
		every raw value to the cloud is expensive and mostly pointless.
		\item \textbf{Resilience.} If the building's internet dies, should the heating stop working?
		\subtle{(Real answer from real deployments: yes, embarrassingly often.)}
		\item \textbf{Privacy.} Occupancy traces reveal who is where, when. Processing locally means
		the data \hilite{never has to leave the building}.
	\end{itemize}
	\vspace{0.4em}
	\centering\small \hilite{Fog computing} adds the layer in between — gateways and local servers
	that pre-process before anything reaches the cloud. \subtle{Active research here at LTU.}
\end{frame}

% ---------------------------------------------------------------- 17. Why simulated
\begin{frame}{Why a \emph{simulated} building?}
	\begin{itemize}
		\setlength\itemsep{0.7em}
		\item Testing autonomous control on a real building is \hilite{slow, costly, unsafe}, and
		impossible to reproduce. \subtle{(``Let's set fire to A-huset and see if the model converges.'')}
		\item This is \textbf{system-level simulation} — the standard industry practice: develop and
		evaluate the environment, sensors, actuators, communication, and control software
		\hilite{as one complete CPS} before touching real hardware.
		\item The bar: model the building \hilite{realistically enough} that your architecture would
		survive contact with a real one — same interfaces, same patterns, same failure handling.
		\item What simulation buys you: \hilite{controlled, repeatable experiments} — including the
		fault-injection and stress tests your grade depends on.
	\end{itemize}
	\vspace{0.4em}
	\centering\small\subtle{Optional: plug in real hardware (Raspberry Pi, ESP32) behind the same
	interfaces — extra evidence of good abstractions, never required.}
\end{frame}

% ---------------------------------------------------------------- 18. Simulator
\begin{frame}{Your physical simulator: realistic enough, not perfect}
	\textbf{You write the physics} — and it must \hilite{respond to actuators} (that's the loop):
	\begin{itemize}
		\setlength\itemsep{0.4em}
		\item \textbf{Fire:} smoke rises and spreads to adjacent rooms; sprinklers knock it down
		\item \textbf{HVAC:} temperature follows outdoor weather, occupancy (body heat), and heating state — \hilite{gradually, not instantly}
		\item \textbf{Air:} CO$_2$ climbs $\sim$40\,ppm per person-hour; ventilation brings it down
		\item \textbf{Occupancy drives everything:} arrive 7--9, lunch 11:30--13, leave 16--18;
		meeting rooms in blocks; weekends empty \subtle{(and it feeds temperature, CO$_2$, lighting, access)}
	\end{itemize}
	\vspace{0.5em}
	\textbf{Four valid approaches:} physics equations \;·\; replay of public datasets
	\subtle{(ASHRAE, Building Data Genome\ldots)} \;·\; hybrid (real data + injected events) \;·\; generative
	\vspace{0.5em}
	\begin{beamercolorbox}[rounded=true,sep=5pt]{block body alerted}
		\color{mainColor}\small\textbf{Capture trends, correlations, and responses — then stop.
		Weeks spent perfecting physics are weeks stolen from architecture, integration, and testing.}
	\end{beamercolorbox}
\end{frame}

% ---------------------------------------------------------------- 25. Pass/fail gate
\begin{frame}{The Pass/Fail gate, spelled out}
	\small Before any grade is discussed, \hilite{all} of these must exist:
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{itemize}
				\item Sensors, actuators, decision-making: \hilite{separate, restartable processes} — no monolith
				\item \hilite{End-to-end}: sensing $\rightarrow$ decision $\rightarrow$ actuation $\rightarrow$
				visible effect $\rightarrow$ next readings
				\item $\geq$1 component makes \hilite{autonomous decisions} from sensor data
				\item A \hilite{data pipeline}: collect, store, serve
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{itemize}
				\item \hilite{Architecture document}: C4 context + container, requirements, interfaces
				\item \hilite{Test plan, executed}: unit + $\geq$1 end-to-end, documented results
				\item A \hilite{dashboard} and a \hilite{written report}
				\item Oral defence: architecture \hilite{from memory}
			\end{itemize}
		\end{column}
	\end{columns}
	\vspace{0.7em}
	\centering\small Missing any one $\Rightarrow$ \textcolor{badRed}{\textbf{U (Fail)}} until fixed —
	\subtle{the gate is binary; the polish comes after.}
\end{frame}

% ---------------------------------------------------------------- 27. Oral exam
\begin{frame}{The oral exam: 30 minutes, you and a whiteboard}
	\begin{columns}[T]
		\begin{column}{0.46\textwidth}
			\textbf{Format \subtle{(individual)}}
			\begin{itemize}
				\item 10 min — live demo of your system
				\item 20 min — \hilite{whiteboard}: draw and explain, \hilite{no slides, no notes}
			\end{itemize}
			\vspace{0.4em}
			\textbf{You must be able to}
			\begin{itemize}
				\item Draw the architecture from memory
				\item Trace a reading from sensor to action
				\item Justify each choice \hilite{against an alternative}
				\item Explain your testing and failure behaviour
			\end{itemize}
		\end{column}
		\begin{column}{0.50\textwidth}
			\textbf{Questions to expect}
			\begin{itemize}
				\item What makes your system \emph{autonomous} rather than \emph{automated}?
				\item What happens if a sensor lies — or the agent process dies?
				\item Why \emph{this} communication pattern?
				\item How would it scale to ten buildings?
			\end{itemize}
			\vspace{0.4em}
			\small\subtle{Be honest about what you don't know — it beats guessing.
			Focus on \emph{why}, not \emph{what}: reasoning, not memorisation.}
		\end{column}
	\end{columns}
\end{frame}

% ---------------------------------------------------------------- 28. Advice
\begin{frame}{How to come out on top (read this twice)}
	\begin{enumerate}
		\setlength\itemsep{0.6em}
		\item \hilite{Close the loop first.} One sensor, one rule, one actuator, end-to-end, in week one
		of building. A walking skeleton you grow beats five polished components that never met.
		\item \hilite{Spend your physics budget wisely.} Realistic enough to learn from — then stop.
		The grade lives in architecture, integration, and evaluation.
		\item \hilite{Kill your own processes.} Crash the agent mid-decision, feed it lying sensors.
		Fault injection is where grades 4 and 5 live \subtle{(dimension D loves evidence)}.
		\item \hilite{Keep the AI boring.} A justified, evaluated anomaly detector outscores an
		unexplained LLM stack. Data-driven $\neq$ deep learning.
		\item \hilite{Let the architecture document live.} Update it as you build — at the oral you
		will redraw it from memory, and that's easy if it was never fiction.
	\end{enumerate}
\end{frame}
